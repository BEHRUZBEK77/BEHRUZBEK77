// 🔧 Custom Hook: useLocalStorage
// localStorage bilan ishlashni osonlashtiradi

import { useState, useEffect } from 'react';

function useLocalStorage(key, initialValue) {
  // localStorage dan qiymatni olish
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(`useLocalStorage error:`, error);
      return initialValue;
    }
  });

  // Yangi qiymat saqlash
  const setValue = (value) => {
    try {
      const valueToStore =
        value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    } catch (error) {
      console.error(`useLocalStorage setValue error:`, error);
    }
  };

  return [storedValue, setValue];
}

// Ishlatish / Usage:
// const [theme, setTheme] = useLocalStorage('theme', 'light');

// 🌙 Dark Mode Toggle — Tailwind CSS + React

import { useState, useEffect } from 'react';

const DarkModeToggle = () => {
  const [isDark, setIsDark] = useState(false);

  useEffect(() => {
    // Saqlangan temani olish
    const savedTheme = localStorage.getItem('theme');
    const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    
    if (savedTheme === 'dark' || (!savedTheme && prefersDark)) {
      setIsDark(true);
      document.documentElement.classList.add('dark');
    }
  }, []);

  const toggleTheme = () => {
    if (isDark) {
      document.documentElement.classList.remove('dark');
      localStorage.setItem('theme', 'light');
      setIsDark(false);
    } else {
      document.documentElement.classList.add('dark');
      localStorage.setItem('theme', 'dark');
      setIsDark(true);
    }
  };

  return (
    <button
      onClick={toggleTheme}
      className="
        relative w-14 h-7 rounded-full transition-colors duration-300
        bg-gray-200 dark:bg-indigo-600
        focus:outline-none focus:ring-2 focus:ring-indigo-500
      "
      aria-label="Toggle dark mode"
    >
      {/* Sliding circle */}
      <span
        className={`
          absolute top-0.5 left-0.5
          w-6 h-6 rounded-full
          bg-white shadow-md
          transition-transform duration-300
          flex items-center justify-center text-sm
          ${isDark ? 'translate-x-7' : 'translate-x-0'}
        `}
      >
        {isDark ? '🌙' : '☀️'}
      </span>
    </button>
  );
};

export default DarkModeToggle;

// 📡 useFetch — Universal Data Fetching Hook

import { useState, useEffect, useCallback } from 'react';

function useFetch(url, options = {}) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const fetchData = useCallback(async () => {
    if (!url) return;

    setLoading(true);
    setError(null);

    try {
      const response = await fetch(url, {
        headers: {
          'Content-Type': 'application/json',
          ...options.headers,
        },
        ...options,
      });

      if (!response.ok) {
        throw new Error(`HTTP Error: ${response.status} ${response.statusText}`);
      }

      const result = await response.json();
      setData(result);
    } catch (err) {
      setError(err.message || 'Noma\'lum xato yuz berdi');
    } finally {
      setLoading(false);
    }
  }, [url]);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  return { data, loading, error, refetch: fetchData };
}

## ❓ Ko'p so'raladigan savollar / FAQ

<details>
<summary><b>🇺🇿 Necha yoshdan va qanday dasturlashni boshladingiz?</b></summary>
<br>
Men bolaligimdan kompyuter va texnologiyalarga qiziqib kelganman. HTML va CSS bilan oddiy sahifalar yaratishdan boshladim. Dastlab YouTube videolari orqali o'rgandim, keyin esa rasmiy dokumentatsiyalarga o'tdim. Hozirda React.js va zamonaviy frontend texnologiyalari bilan ishlayapman. Yoshim 13 da bo'lsa ham, har kuni kamida 3-4 soat kod yozaman.
</details>

<details>
<summary><b>🇬🇧 How do you learn so fast at such a young age?</b></summary>
<br>
Consistency is my secret weapon. I code every single day, even if it's just for 30 minutes. I learn by doing — I don't just watch tutorials, I build things. When I get stuck, I try to solve it myself first, then look up the solution. Every error message is a learning opportunity. I also believe in starting small: build a button, then a card, then a full page, then a full app.
</details>

<details>
<summary><b>🇷🇺 Какой совет вы дадите начинающим разработчикам?</b></summary>
<br>
Главное — не сравнивайте себя с другими. У каждого свой темп. Сосредоточьтесь на основах: HTML, CSS, JavaScript — изучите их хорошо, прежде чем переходить к фреймворкам. Стройте что-то каждый день. Не бойтесь делать ошибки — это нормально. Используйте Git с самого начала. И самое важное — получайте удовольствие от процесса!
</details>

<details>
<summary><b>🇺🇿 Kelajakda qanday loyihalar qilishni rejalashtirmoqdasiz?</b></summary>

- 🤝 **Portfolio v2** — Yangi dizayn va yangi texnologiyalar bilan yangilangan portfolio
- 📱 **Mobile App** — React Native bilan birinchi mobil ilovam

</details>

<details>
<summary><b>🇺🇿 Open-source loyihalarga qo'shilasizmi?</b></summary>

Ammo kelajakda, kerakli tajriba va bilimlarni to'plaganimdan so'ng, open-source hamjamiyatga faol hissa qo'shishni qat'iy rejalashtirmoqdaman. Bu mening asosiy maqsadlarimdan biri!

## ⚙️ GitHub Actions sozlamasi / GitHub Actions Setup

<details>
<summary><b>🤖 metrics.yml — Metrikalarni avtomatik yangilash</b></summary>
<br>

Ushbu workflow-ni `.github/workflows/metrics.yml` ga saqlang:

```yaml
# ═══════════════════════════════════════════════════════════
# GitHub Metrics — Automated SVG Generation
# Runs daily at midnight UTC and on manual trigger
# ═══════════════════════════════════════════════════════════

name: 📊 GitHub Metrics

on:
  # Har kecha yarim tunda yangilanadi
  schedule:
    - cron: '0 0 * * *'
  
  # Qo'lda ishlatish imkoniyati
  workflow_dispatch: {}
  
  # Main branchga push bo'lsa
  push:
    branches: [main, master]

jobs:
  github-metrics:
    name: Generate GitHub Metrics
    runs-on: ubuntu-latest
    
    permissions:
      contents: write
    
    steps:
      - name: 📊 Generate Metrics SVG
        uses: lowlighter/metrics@latest
        with:
          # GitHub token (Settings > Developer settings > Tokens)
          token: ${{ secrets.METRICS_TOKEN }}
          
          # GitHub username
          user: BEHRUZBEK77
          
          # Chiqish fayl nomi
          filename: github-metrics.svg
          
          # Template va o'lcham
          template: classic
          base: header, activity, community, repositories, metadata
          
          # ──── PLUGINLAR ────
          
          # Tillar statistikasi
          plugin_languages: yes
          plugin_languages_ignored: html, css
          plugin_languages_limit: 8
          plugin_languages_sections: most-used, recently-used
          
          # Kunlik odatlar tahlili
          plugin_habits: yes
          plugin_habits_from: 200
          plugin_habits_days: 14
          plugin_habits_charts: yes
          
          # Faollik tarixi
          plugin_activity: yes
          plugin_activity_limit: 5
          plugin_activity_days: 14
          
          # Calendar heatmap
          plugin_calendar: yes
          plugin_calendar_limit: 1
          
          # Isocal grafik
          plugin_isocalendar: yes
          plugin_isocalendar_duration: half-year
          
          # Reaktsiyalar
          plugin_reactions: yes
          plugin_reactions_limit: 200
          
          # Notable kontribusiyalar
          plugin_notable: yes
          plugin_notable_filter: stars:>100
```

**Sozlash bosqichlari:**

1. **Token yaratish:** `GitHub Settings` → `Developer settings` → `Personal access tokens` → `Tokens (classic)`
2. **Kerakli ruxsatlar:** `repo`, `read:user`, `read:org`
3. **Secretni qo'shish:** `Repository Settings` → `Secrets and variables` → `Actions` → `New repository secret`
   - **Name:** `METRICS_TOKEN`
   - **Value:** Yaratilgan tokeningiz
4. **Workflow faylni yaratish:** Yuqoridagi YAML-ni `.github/workflows/metrics.yml` ga saqlang
5. **Ishga tushirish:** `Actions` tab → `GitHub Metrics` → `Run workflow`

</details>
